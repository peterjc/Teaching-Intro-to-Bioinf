\subsection{Dynamic Programming}
  \begin{frame}
  \frametitle{Our Goal}
  \begin{itemize}
    \item<1-> We aim to align the words
    \begin{itemize}
      \item<1-> \texttt{COELACANTH}
      \item<1-> \texttt{PELICAN}
    \end{itemize}
    \item<2-> Each identical letter (match) scores +1
    \item<2-> Each different letter (mismatch) scores -1
    \item<2-> Each gap scores -1
    \item<3-> \emph{All sequence alignment is maximisation of an alignment score} - a mathematical operation.
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Algorithms}
  \begin{itemize}
    \item Global: Needleman-Wunsch (as in example)
    \item Local: Smith-Waterman (differs from example)
    \item Biological information encapsulated \emph{only} in the scoring scheme (matches, mismatches, gaps)
    \item NW/SW are \emph{guaranteed} to find the optimal match \emph{with respect to the scoring system being used}
    \item BUT the optimal alignment is a biological approximation: no scoring scheme encapsulates biological ``truth''
    \item Any pair of sequences can be aligned: finding meaning is up to you
  \end{itemize}
\end{frame}
